    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode dump = new ListNode(0, head);
        ListNode slow = dump, fast = head;

        for (int i = 0; i < n; i++) {
            fast = fast.next;
        }

        while (fast != null) {
            slow = slow.next;
            fast = fast.next;
        }

        slow.next = slow.next.next;

        return dump.next;

    }
