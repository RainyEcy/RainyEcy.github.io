public class TwoSumForArray {

    /**

    public int[] twoSum(int[] numbers, int target) {
        for (int left = 0, right = numbers.length - 1; left < right; ) {
            int sum = numbers[left] + numbers[right];
            if (target == sum) {
                return new int[]{++left, ++right};
            } else if (target > sum) {
                left++;
            } else {
                right--;
            }
        }
        return new int[]{-1, -1};
    }
